```bash
export PATH=/tmp/go/bin:$PATH
GOOS=linux GOARCH=arm GOARM=7 CGO_ENABLED=0 \
  go build -ldflags="-s -w" -o luckfox_network main.go
```

```python
ser = serial.Serial('COM14', 115200, timeout=3)
```
